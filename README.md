## Identificação de complexidade de cada método utilizado.

addFirst(value: T) - O(1)

addLast(value: T) - O(1)

clear() - O(1)

contains(value: T):boolean - O(n)

getFirst(): T - O(1)

getLast(): T - O(1)

isEmpty():boolean - O(1)

removeFirst(): T - O(1)

removeLast(): T - O(1)

size() - O(1)

toArray():Object[0...*] - O(n)

get(index: int):T - O(n)

remove(index: int):T - O(n)

add(index: int, value:T) - O(n)

set(index:int, value: T) - O(n)
